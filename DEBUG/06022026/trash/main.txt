import asyncio
import os
import sys
from aiogram import Bot, Dispatcher
from aiogram.client.default import DefaultBotProperties
from aiogram.enums import ParseMode
from aiogram.fsm.storage.memory import MemoryStorage  # ‚Üê –î–û–ë–ê–í–õ–ï–ù –ò–ú–ü–û–†–¢
from dotenv import load_dotenv
from handlers.start import router as start_router
from services.price_watcher import PriceWatcher
from utils.logger import logger
from utils.redis_client import redis_client
import logging

logging.basicConfig(level=logging.DEBUG)
load_dotenv()

async def main():
    # –ü–æ–¥–∫–ª—é—á–µ–Ω–∏–µ –∫ Redis
    try:
        await redis_client.connect()
    except Exception as e:
        logger.error(f"–û—à–∏–±–∫–∞ –ø–æ–¥–∫–ª—é—á–µ–Ω–∏—è –∫ Redis: {e}")
        logger.info("–ü—Ä–æ–¥–æ–ª–∂–∞—é —Ä–∞–±–æ—Ç—É –±–µ–∑ –∫—ç—à–∏—Ä–æ–≤–∞–Ω–∏—è...")
        # –ü—Ä–æ–¥–æ–ª–∂–∞–µ–º —Ä–∞–±–æ—Ç—É –±–µ–∑ –∫—ç—à–∞ (–Ω–µ –æ—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ–º –±–æ—Ç)
        # sys.exit(1) ‚Üê –£–î–ê–õ–ï–ù–û
    
    # –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è –±–æ—Ç–∞
    bot = Bot(
        token=os.getenv("BOT_TOKEN"),
        default=DefaultBotProperties(parse_mode=ParseMode.HTML)
    )
    
    # –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è –Ω–∞–±–ª—é–¥–∞—Ç–µ–ª—è –∑–∞ —Ü–µ–Ω–∞–º–∏
    price_watcher = PriceWatcher(bot)
    
    # ‚Üê –ó–ê–ú–ï–ù–ê –≠–¢–û–ô –°–¢–†–û–ö–ò:
    # –ë–´–õ–û: dp = Dispatcher()
    # –°–¢–ê–õ–û:
    dp = Dispatcher(storage=MemoryStorage())  # ‚Üê MemoryStorage –¥–ª—è —Ä–∞–±–æ—Ç—ã —Å —Å–æ—Å—Ç–æ—è–Ω–∏—è–º–∏
    
    dp.include_router(start_router)
    
    logger.info("üöÄ –ë–æ—Ç –∑–∞–ø—É—â–µ–Ω!")
    
    # –ó–∞–ø—É—Å–∫ –Ω–∞–±–ª—é–¥–∞—Ç–µ–ª—è –≤ —Ñ–æ–Ω–µ
    watcher_task = asyncio.create_task(price_watcher.start())
    
    try:
        await dp.start_polling(bot)
    finally:
        # –û—Å—Ç–∞–Ω–æ–≤–∫–∞ –Ω–∞–±–ª—é–¥–∞—Ç–µ–ª—è
        price_watcher.running = False
        watcher_task.cancel()
        
        # –ó–∞–∫—Ä—ã—Ç–∏–µ —Å–æ–µ–¥–∏–Ω–µ–Ω–∏—è —Å Redis
        await redis_client.close()
        logger.info("‚úÖ Redis —Å–æ–µ–¥–∏–Ω–µ–Ω–∏–µ –∑–∞–∫—Ä—ã—Ç–æ")

if __name__ == "__main__":
    asyncio.run(main())
